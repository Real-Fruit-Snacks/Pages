NAME
       Text::LineFold~[ja] -

SYNOPSIS
           use Text::LineFold;
           $lf = Text::LineFold->new();

           #
           $folded = $lf->fold($string, 'PLAIN');
           $indented = $lf->fold(' ' x 8, ' ' x 4, $string);

           #
           $unfolded = $lf->unfold($string, 'FIXED');

DESCRIPTION
       Text::LineFold RFC 3676  flowed

       new ([KEY => VALUE, ...])
            KEY => VALUE  config

       $self->config (KEY)
       $self->config ([KEY => VAL, ...])
             KEY => VALUE

           Charset => CHARSET
                 MIME::Charset~[ja]  "UTF-8"

           Language => LANGUAGE
               Charset /  "XX" "Context" in Unicode::LineBreak~[ja]

           Newline => STRING
                 "\n"

           OutputCharset => CHARSET
               fold()/unfold()  MIME::Charset~[ja]  "_UNICODE_"  Unicode
               Charset

           TabSize => NUMBER
                0  8

           BreakIndent
           CharMax
           ColMax
           ColMin
           ComplexBreaking
           EAWidth
           HangulAsAL
           LBClass
           LegacyCM
           Prep
           Urgent
               "" in Unicode::LineBreak~[ja]

       $self->fold (STRING, [METHOD])
       $self->fold (INITIAL_TAB, SUBSEQUENT_TAB, STRING, ...)
             STRING  Newline  TabSize

           METHOD

           "FIXED"
               ">"

           "FLOWED"
               RFC 3676  "Format=Flowed; DelSp=Yes"

           "PLAIN"

           "wrap()" in Text::Wrap  INITIAL_TAB  SUBSEQUENT_TAB

       $self->unfold (STRING, METHOD)
            STRING

           METHOD

           "FIXED"
                ">"

           "FLOWED"
               RFC 3676  "Format=Flowed; DelSp=Yes"

           "FLOWEDSP"
               RFC 3676  "Format=Flowed; DelSp=No"
