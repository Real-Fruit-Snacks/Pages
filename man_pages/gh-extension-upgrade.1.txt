NAME
       gh-extension-upgrade - Upgrade installed extensions

SYNOPSIS
       gh extension upgrade {<name> | --all} [flags]

OPTIONS
       --all  Upgrade all extensions

       --dry-run
              Only display upgrades

       --force
              Force upgrade extension
