NAME
       timespec - interval for a timer with nanosecond precision

LIBRARY
       Standard C library (libc)

SYNOPSIS
       #include <time.h>

       struct itimerspec {
           struct timespec  it_interval;  /* Interval for periodic timer */
           struct timespec  it_value;     /* Initial expiration */
       };
